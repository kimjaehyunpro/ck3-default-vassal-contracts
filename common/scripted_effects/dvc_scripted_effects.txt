dvc_apply_defaults_effect = {
    debug_log = "DVC EFFECT: apply_defaults START"
    save_scope_as = dvc_liege
    if = {
        limit = { scope:dvc_liege.var:dvc_apply_indirect = 1 }
        debug_log = "DVC EFFECT: Applying to ALL vassals (direct + indirect)..."
        every_vassal_or_below = {
            dvc_apply_contract_to_single_vassal_effect = yes
        }
        debug_log = "DVC EFFECT: Finished all vassals."
    }
    else = {
        debug_log = "DVC EFFECT: Applying to DIRECT vassals only..."
        every_vassal = {
            dvc_apply_contract_to_single_vassal_effect = yes
        }
        debug_log = "DVC EFFECT: Finished direct vassals."
    }
    debug_log = "DVC EFFECT: apply_defaults END"
}

dvc_apply_contract_to_single_vassal_effect = {
    debug_log = "DVC APPLY: Processing one vassal"
    if = {
        limit = { vassal_contract_has_modifiable_obligations = yes }
        debug_log = "DVC APPLY: Has modifiable obligations - proceeding."
        if = {
            limit = { scope:dvc_liege.var:dvc_default_tax = 0 }
            debug_log = "DVC APPLY: Setting TAX to exempt."
            vassal_contract_set_obligation_level = { type = feudal_government_taxes level = feudal_tax_exempt_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_tax = 1 }
            debug_log = "DVC APPLY: Setting TAX to low."
            vassal_contract_set_obligation_level = { type = feudal_government_taxes level = feudal_tax_low_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_tax = 2 }
            debug_log = "DVC APPLY: Setting TAX to normal."
            vassal_contract_set_obligation_level = { type = feudal_government_taxes level = feudal_tax_normal_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_tax = 3 }
            debug_log = "DVC APPLY: Setting TAX to high."
            vassal_contract_set_obligation_level = { type = feudal_government_taxes level = feudal_tax_high_level }
        }
        if = {
            limit = { scope:dvc_liege.var:dvc_default_levies = 0 }
            debug_log = "DVC APPLY: Setting LEVIES to exempt."
            vassal_contract_set_obligation_level = { type = feudal_government_levies level = feudal_levies_exempt_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_levies = 1 }
            debug_log = "DVC APPLY: Setting LEVIES to low."
            vassal_contract_set_obligation_level = { type = feudal_government_levies level = feudal_levies_low_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_levies = 2 }
            debug_log = "DVC APPLY: Setting LEVIES to normal."
            vassal_contract_set_obligation_level = { type = feudal_government_levies level = feudal_levies_normal_level }
        }
        else_if = {
            limit = { scope:dvc_liege.var:dvc_default_levies = 3 }
            debug_log = "DVC APPLY: Setting LEVIES to high."
            vassal_contract_set_obligation_level = { type = feudal_government_levies level = feudal_levies_high_level }
        }
        if = {
            limit = {
                government_has_flag = government_is_feudal
                can_have_confederate_partition_succession_law_trigger = yes
            }
            if = {
                limit = { scope:dvc_liege.var:dvc_forced_partition = 1 }
                debug_log = "DVC APPLY: Setting FORCED PARTITION on."
                vassal_contract_set_obligation_level = { type = succession_rights level = succession_rights_forced_partition }
            }
            else = {
                debug_log = "DVC APPLY: Setting FORCED PARTITION off."
                vassal_contract_set_obligation_level = { type = succession_rights level = succession_rights_default }
            }
        }
        else = {
            debug_log = "DVC APPLY: Skipped forced partition - not feudal or cannot have confederate partition."
        }
    }
    else = {
        debug_log = "DVC APPLY: No modifiable obligations - skipped."
    }
}
