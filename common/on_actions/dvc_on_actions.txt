on_vassal_gained = {
    effect = {
        debug_log = "DVC ON_ACTION: on_vassal_gained fired."
        if = {
            limit = {
                is_ai = no
                var:dvc_auto_apply = 1
            }
            debug_log = "DVC ON_ACTION: on_vassal_gained - auto-apply ON, applying to vassal."
            save_scope_as = dvc_liege
            scope:vassal = {
                dvc_apply_contract_to_single_vassal_effect = yes
            }
        }
    }
}

on_title_gain = {
    effect = {
        debug_log = "DVC ON_ACTION: on_title_gain fired."
        if = {
            limit = {
                exists = liege
                liege = {
                    is_ai = no
                    var:dvc_auto_apply = 1
                }
            }
            debug_log = "DVC ON_ACTION: on_title_gain - direct vassal of player, applying defaults."
            liege = { save_scope_as = dvc_liege }
            dvc_apply_contract_to_single_vassal_effect = yes
        }
    }
}
