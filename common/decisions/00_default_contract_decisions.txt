toggle_universal_contract_rule = {
    is_shown = {
        is_landed = yes
    }

    effect = {
        if = {
            limit = { has_character_flag = universal_contract_rule_active }
            remove_character_flag = universal_contract_rule_active
            custom_tooltip = toggle_rule_disabled_tt
        }
        else = {
            add_character_flag = universal_contract_rule_active
            custom_tooltip = toggle_rule_enabled_tt
        }
    }
}
