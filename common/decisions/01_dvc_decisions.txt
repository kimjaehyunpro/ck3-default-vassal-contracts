dvc_open_settings_decision = {

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
    }

    desc = dvc_open_settings_decision_desc

    is_shown = {
        is_ai = no
        is_ruler = yes
    }

    # No is_valid block — "always = yes" renders as
    # "always has no localization" in the tooltip, so we omit it entirely.

    effect = {
        debug_log = "DVC DECISION: Player opened DVC settings menu."

        # Initialize variables only on first ever use.
        # Safe to leave in — if variables exist this block is skipped.
        if = {
            limit = { NOT = { exists = var:dvc_initialized } }
            debug_log = "DVC DECISION: First-time init — setting default variables."
            set_variable = { name = dvc_initialized      value = 1 }
            set_variable = { name = dvc_default_tax      value = 0 } # 0=exempt 1=low 2=normal 3=high
            set_variable = { name = dvc_default_levies   value = 4 }
            set_variable = { name = dvc_forced_partition         value = 1 } # 0=off 1=on
            set_variable = { name = dvc_default_special_contract value = 0 } # 0=none 1=scutage 2=march 3=castellan 4=palatinate
            set_variable = { name = dvc_fortification_rights     value = 0 } # 0=none 1=granted
            set_variable = { name = dvc_coinage_rights           value = 0 } # 0=none 1=granted
            set_variable = { name = dvc_war_declaration_rights   value = 0 } # 0=default 1=allowed
            set_variable = { name = dvc_council_rights           value = 0 } # 0=none 1=guaranteed
            set_variable = { name = dvc_title_revocation_rights  value = 0 } # 0=default 1=protected
            set_variable = { name = dvc_apply_indirect           value = 1 } # Option B: 0=off 1=on
            set_variable = { name = dvc_auto_apply       value = 1 } # Auto-apply on new vassal: 0=off 1=on
        }

        trigger_event = { id = dvc.001 }
    }

    ai_will_do = {
        base = 0
    }

    ai_check_interval = 0
}
